// Check BitLocker status via Powershell
ps, err := exec.LookPath("powershell.exe")
if err != nil {
	myDisk.BitLockerStatus = "Unknown"
} else {
	cmd := exec.Command(ps, fmt.Sprintf("(New-Object -ComObject Shell.Application).NameSpace('%s').Self.ExtendedProperty('System.Volume.BitLockerProtection')", myDisk.Label))
	out, err := cmd.Output()
	if err != nil {
		log.Printf("[ERROR]: error executing powershell to check bitlocker status: %v", err)
	}
	blStatus, err := strconv.Atoi(fmt.Sprintf("%s", out))
	if err != nil {
		log.Printf("[ERROR]: error parsing powershell bitlocker query: %v", err)
	}

	switch blStatus {
	case 1, 3, 5:
		myDisk.BitLockerStatus = "Encrypted"
	default:
		myDisk.BitLockerStatus = "Unencrypted"
	}
}
